***Query usada no MySQL Workbench durante a avaliação***


# Criação do banco de dados
CREATE DATABASE bdprogweb;
USE bdprogweb;

# Criação da tabela produto
CREATE TABLE produto (
	id_produto INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
	descricao VARCHAR(1000),
	preco FLOAT
);

# Comando SELECT para auxiliar no exercício 1
SELECT * FROM produto;

# Comando TRUNCATE para apagar inserções erradas no banco de dados
TRUNCATE TABLE produto;
